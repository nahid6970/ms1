            self.pty.spawn("powershell.exe", cmdline=cmdline, cwd=path)
        
        # Start background thread to read PTY output and broadcast via WebSocket
        self.read_thread = threading.Thread(target=self._read_loop, daemon=True)
        self.read_thread.start()
        
    def _read_loop(self):
        while self.pty.isalive():
            try:
                # Non-blocking read with small sleep to reduce latency
                data = self.pty.read(blocking=False)
                if data:
                    with self.history_lock:
                        self.history += data
                        if len(self.history) > 100000:
                            self.history = self.history[-100000:]
                    # Broadcast to all connected WebSocket clients
                    with self.sids_lock:
                        sids = list(self.connected_sids)
                    for sid in sids:
                        try:
                            socketio.emit('pty-output', {'data': data}, room=sid, namespace='/pty')
                        except Exception:
                            pass
                else:
                    import time
                    time.sleep(0.005)  # 5ms poll to keep CPU low but responsive
            except Exception:
                break
        # Notify all clients that session ended
        with self.sids_lock:
            sids = list(self.connected_sids)
        for sid in sids:
            try:
                socketio.emit('pty-output', {'eof': True}, room=sid, namespace='/pty')
            except Exception:
                pass

    def write(self, data):
        if self.pty.isalive():
            self.pty.write(data)

    def resize(self, cols, rows):
        if self.cols != cols or self.rows != rows:
            self.cols = cols
            self.rows = rows
            if self.pty.isalive():
                self.pty.set_size(cols, rows)

    def add_sid(self, sid):
        with self.sids_lock:
            self.connected_sids.add(sid)

    def remove_sid(self, sid):
        with self.sids_lock:
            self.connected_sids.discard(sid)

    def get_history(self):
        with self.history_lock:
            return self.history

def load_projects_config():
    if os.path.exists(PROJECTS_FILE):
        try:
            with open(PROJECTS_FILE, "r", encoding="utf-8") as f:
                projs = json.load(f)
                # Sanitize bookmarks to be dicts
                for p in projs:
                    if "bookmarks" not in p:
                        p["bookmarks"] = []
                    else:
                        sanitized = []
                        for bm in p["bookmarks"]:
                            if isinstance(bm, str):
                                sanitized.append({"command": bm, "global": False})
                            elif isinstance(bm, dict):
                                sanitized.append({
                                    "command": bm.get("command", ""),
                                    "global": bm.get("global", False)
                                })
                        p["bookmarks"] = sanitized
                return projs
        except Exception as e:
            print(f"Error reading projects config: {e}")
    return []

        if "pathColor" not in card_theme:
            card_theme["pathColor"] = "#94a3b8"
        
        with sessions_lock:
            is_active = name in active_sessions and active_sessions[name].pty.isalive()
            
        bookmarks = p.get("bookmarks", [])

            "is_active": is_active,
            "pinned": pinned,
            "theme": theme,
            "cardTheme": card_theme,
            "bookmarks": bookmarks
        })
    return projects

    import datetime
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    
    if is_clipboard:
        filename = f"pasted_image_{timestamp}.png"

        # For dragged files, preserve their original name to avoid confusion
        name_part, ext_part = os.path.splitext(original_filename)
        name_part = os.path.basename(name_part)  # Sanitize name
        if not ext_part:
            ext_part = ".png"
        filename = f"{name_part}_{timestamp}{ext_part}"
    
    dest_path = os.path.join(proj["path"], filename)
    try:
        file.save(dest_path)
        return jsonify({
            "status": "success",
            "path": dest_path.replace("\\", "/"),
            "filename": filename
        })
    except Exception as e:
        return jsonify({"error": f"Failed to save image: {str(e)}"}), 500

@socketio.on('join-session', namespace='/pty')
def ws_join_session(data):
    """Client wants to attach to a project session."""
    project = data.get('project')
    if not project:
        emit('error', {'msg': 'Project name required'})
        return
        
    projects = scan_projects()
    proj_details = next((p for p in projects if p["name"] == project), None)
    if not proj_details:
        emit('error', {'msg': 'Project not found'})
        return
        
    with sessions_lock:
        if project not in active_sessions or not active_sessions[project].pty.isalive():
            active_sessions[project] = TerminalSession(project, proj_details["path"])
        session = active_sessions[project]
        session.add_sid(request.sid)
        
    # Send accumulated history immediately
    history = session.get_history()
    if history:
        emit('pty-output', {'data': history})
        
    emit('session-ready', {'name': project, 'path': proj_details["path"]})

@socketio.on('pty-input', namespace='/pty')
def ws_pty_input(data):
    """Receive keystrokes from browser and write directly to PTY."""
    project = data.get('project')
    input_data = data.get('data', '')
    if not project:
        return
        
    with sessions_lock:
        session = active_sessions.get(project)
    if session and session.pty.isalive():
        session.write(input_data)

@socketio.on('resize', namespace='/pty')

        // Fetch project workspaces from Flask server
        async function fetchProjects(refresh = false) {
            try {
                const url = refresh ? '/api/projects?refresh=true' : '/api/projects';
                const response = await fetch(url);
                projects = await response.json();
                
                // Keep active project details updated with latest theme
                if (activeProject) {
                    const activeDetails = projects.find(p => p.name === activeProject);
                    if (activeDetails && term) {
                        term.options.theme = activeDetails.theme;
                    }
                }
                
                handleSearch(); // Triggers render with current search filter
            } catch (err) {
                console.error("Error fetching projects:", err);
            }
        }

            container.innerHTML = '';

            // Bind container drag events once
            if (!container.dataset.dragBound) {
                container.dataset.dragBound = 'true';
                container.ondragover = (e) => {
                    e.preventDefault();
                    const draggingCard = document.querySelector('.dragging');
                    if (!draggingCard) return;
                    
                    const isDraggingPinned = draggingCard.dataset.pinned === 'true';
                    const afterElement = getDragAfterElement(container, e.clientY);
                    
                    // Find the first normal (unpinned) card in the list to act as the boundary
                    const firstNormalCard = [...container.querySelectorAll('.project-card')].find(c => c !== draggingCard && c.dataset.pinned === 'false');
                    
                    if (isDraggingPinned) {
                        // A pinned card can only be placed in the pinned section (above the first normal card)
                        if (afterElement && afterElement.dataset.pinned === 'true') {
                            container.insertBefore(draggingCard, afterElement);
                        } else {
                            if (firstNormalCard) {
                                container.insertBefore(draggingCard, firstNormalCard);
                            } else {
                                container.appendChild(draggingCard);
                            }
                        }
                    } else {
                        // A normal card can only be placed in the normal section (at or below the first normal card)
                        if (afterElement && afterElement.dataset.pinned === 'false') {
                            container.insertBefore(draggingCard, afterElement);
                        } else {
                            if (firstNormalCard) {
                                container.insertBefore(draggingCard, firstNormalCard);
                            } else {
                                container.appendChild(draggingCard);
                            }
                        }
                    }
                };
            }

            const filtered = projects.filter(p => 
                p.name.toLowerCase().includes(searchQuery) || 
                p.path.toLowerCase().includes(searchQuery)
            );

            if (filtered.length === 0) {
                container.innerHTML = `
                    <div style="text-align: center; color: var(--text-secondary); padding: 30px 10px; font-size: 0.9rem;">
                        No workspaces found.
                    </div>
                `;
                return;
            }

            filtered.forEach(proj => {
                const card = document.createElement('div');
                card.className = `project-card ${activeProject === proj.name ? 'active' : ''} ${proj.is_active ? 'running' : ''}`;
                
                // Enable drag and drop reordering
                card.setAttribute('draggable', 'true');
                card.setAttribute('data-name', proj.name);
                card.dataset.pinned = proj.pinned ? 'true' : 'false';
                
                card.ondragstart = (e) => {
                    setTimeout(() => card.classList.add('dragging'), 0);
                    e.dataTransfer.effectAllowed = 'move';
                };
                
                card.ondragend = () => {
                    card.classList.remove('dragging');
                    saveProjectsOrder();
                };
                
                // Apply card customization style
                const isCardActive = activeProject === proj.name;
                if (proj.cardTheme) {
                    card.style.backgroundColor = proj.cardTheme.bgColor;
                    card.style.borderColor = isCardActive ? proj.cardTheme.accentColor : 'var(--border-color)';
                    if (isCardActive) {
                        card.style.boxShadow = `0 0 15px ${proj.cardTheme.accentColor}40`;
                    } else {
                        card.style.boxShadow = '';
                    }
                    if (proj.is_active) {
                        card.style.borderLeft = `4px solid ${proj.cardTheme.accentColor}`;
                    } else {
                        card.style.borderLeft = '';
                    }
                }
                
                card.onclick = () => selectProject(proj.name);

                // Title bar container inside card
                const titleContainer = document.createElement('div');
                titleContainer.className = 'project-title-container';

                const title = document.createElement('span');
                title.className = 'project-title';
                if (proj.cardTheme) {
                    title.style.color = proj.cardTheme.textColor;
                }

                const titleText = document.createTextNode(proj.name);
                title.appendChild(titleText);

                // Add Pin icon suffix if workspace is pinned
                if (proj.pinned) {
                    const pin = document.createElement('span');
                    pin.innerText = ' 📌';
                    title.appendChild(pin);
                }
                titleContainer.appendChild(title);

                card.appendChild(titleContainer);

                const path = document.createElement('div');
                path.className = 'project-path';
                path.innerText = proj.path;
                if (proj.cardTheme) {
                    path.style.color = proj.cardTheme.pathColor;
                }
                card.appendChild(path);

                // Card actions container (Action Dots Pattern)
                const cardActions = document.createElement('div');
                cardActions.className = 'card-actions';

                // Yellow (pinned) / Grey (unpinned) Action Dot
                const pinDot = document.createElement('button');
                pinDot.className = `action-dot ${proj.pinned ? 'yellow' : 'grey'}`;
                pinDot.title = proj.pinned ? 'Unpin Workspace' : 'Pin Workspace';
                pinDot.onclick = (e) => {
                    e.stopPropagation();
                    togglePinProject(proj);
                };
                cardActions.appendChild(pinDot);

                // Blue (Customize) Action Dot
                const customizeDot = document.createElement('button');
                customizeDot.className = 'action-dot blue';
                customizeDot.title = 'Customize Style & Theme';
                customizeDot.onclick = (e) => {
                    e.stopPropagation();
                    contextProject = proj;
                    openThemeModal();
                };
                cardActions.appendChild(customizeDot);

                // Red (Delete) Action Dot
                const deleteDot = document.createElement('button');
                deleteDot.className = 'action-dot red';
                deleteDot.title = 'Remove Workspace';
                deleteDot.onclick = (e) => {
                    e.stopPropagation();
                    deleteProject(proj.name);
                };
                cardActions.appendChild(deleteDot);

                card.appendChild(cardActions);

                container.appendChild(card);
            });
        }

        async function togglePinProject(project) {
            try {
                const response = await fetch('/api/projects/customize', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        name: project.name,
                        pinned: !project.pinned
                    })
                });

                if (response.ok) {
                    projects = await response.json();
                    handleSearch();
                }
            } catch (err) {
                console.error("Error pinning project:", err);
            }
        }

        // Switch workspace project
        async function selectProject(projectName) {
            if (activeProject === projectName) return;
            
            activeProject = projectName;
            handleSearch();

            // Toggle placeholders
            document.getElementById('no-workspace-placeholder').style.display = 'none';
            document.getElementById('workspace-header').style.display = 'flex';
            document.getElementById('terminal-wrapper').style.display = 'flex';

            try {
                const response = await fetch(`/api/session/${projectName}`, { method: 'POST' });
                const data = await response.json();
                
                document.getElementById('active-project-name').innerText = data.name;
                document.getElementById('active-project-path').innerText = data.path;
                
                document.getElementById('active-project-branch').style.display = 'none';

                // Retrieve project config details to fetch its theme
                const projDetails = projects.find(p => p.name === projectName) || data;
                initTerminal(projectName, projDetails.theme);
                updateBookmarksDropdown();
            } catch (err) {
                console.error("Error setting up session:", err);
            }
        }

        // Initialize xterm.js terminal instance with WebSocket
        function initTerminal(projectName, projectTheme = null) {
            // Disconnect previous socket if any
            if (socket) {
                socket.disconnect();
                socket = null;
            }

            const termDiv = document.getElementById('terminal-parent');
            termDiv.innerHTML = ''; 

            // Standard terminal color config
            const defaultTheme = {
                background: '#000000',
                foreground: '#d1d5db',
                cursor: '#3b82f6',
                black: '#1e293b',
                red: '#ef4444',
                green: '#10b981',
                yellow: '#f59e0b',
                blue: '#3b82f6',
                magenta: '#d946ef',
                cyan: '#06b6d4',
                white: '#f1f5f9',
                brightBlack: '#475569',
                brightRed: '#f87171',
                brightGreen: '#34d399',
                brightYellow: '#fbbf24',
                brightBlue: '#60a5fa',
                brightMagenta: '#f472b6',
                brightCyan: '#22d3ee',
                brightWhite: '#ffffff'
            };

            // Merge project customized theme options
            const themeOptions = projectTheme ? { ...defaultTheme, ...projectTheme } : defaultTheme;

            term = new Terminal({
                cursorBlink: true,
                cursorStyle: 'bar',
                fontSize: 14,
                fontFamily: '"Fira Code", Courier, monospace',
                theme: themeOptions
            });

            fitAddon = new FitAddon.FitAddon();
            term.loadAddon(fitAddon);
            term.open(termDiv);
            
            // Register Resize Observer to follow any future container resizing
            const resizeObserver = new ResizeObserver(() => {
                try {
                    fitAddon.fit();
                } catch (e) {
                    // Ignore transient layout fits
                }
            });
            resizeObserver.observe(termDiv);

            term.onResize(size => {
                if (socket && socket.connected) {
                    socket.emit('resize', { project: projectName, cols: size.cols, rows: size.rows });
                }
            });

            // Delay initial fit slightly to ensure browser has computed container layout sizes
            setTimeout(() => {
                try {
                    fitAddon.fit();
                    if (socket && socket.connected) {
                        socket.emit('resize', { project: projectName, cols: term.cols, rows: term.rows });
                    }
                } catch (e) {
                    console.error("Initial terminal fit error:", e);
                }
            }, 100);

            // Handle Ctrl+C (copy) and Ctrl+V (paste) behaviours
            term.attachCustomKeyEventHandler(e => {
                // Ctrl+C: copy selected text
                if (e.ctrlKey && (e.key === 'c' || e.code === 'KeyC')) {
                    if (term.hasSelection()) {
                        navigator.clipboard.writeText(term.getSelection());
                        return false; 
                    }
                }
                
                // Ctrl+V: block raw Ctrl+V character (0x16) from key events.
                if (e.ctrlKey && (e.key === 'v' || e.code === 'KeyV')) {
                    return false;
                }
                return true;
            });

            // ── WebSocket Input: Direct emit, zero HTTP overhead ──
            term.onData(data => {
                // Filter out Device Attributes (DA) queries sent back by xterm.js
                if (data.startsWith('\x1b[?') && data.endsWith('c')) {
                    return;
                }
                // Filter out Operating System Command (OSC) responses
                if (data.startsWith('\x1b]')) {
                    return;
                }
                if (socket && socket.connected) {
                    socket.emit('pty-input', { project: projectName, data: data });
                }
            });

            // ── WebSocket Connection ──
            socket = io({ path: '/socket.io' });

            socket.on('connect', () => {
                console.log('WebSocket connected');
                // Join the PTY namespace and request session
                const ptySocket = io('/pty');
                
                ptySocket.on('connect', () => {
                    console.log('PTY namespace connected');
                    ptySocket.emit('join-session', { project: projectName });
                });
                
                ptySocket.on('session-ready', (data) => {
                    console.log('Session ready:', data);
                });
                
                ptySocket.on('pty-output', (msg) => {
                    if (msg.data) {
                        term.write(msg.data);
                    } else if (msg.eof) {
                        term.write("\r\n\x1b[31m[Shell Session Ended. Refresh or re-select to restart]\x1b[0m\r\n");
                        ptySocket.disconnect();
                    }
                });
                
                ptySocket.on('error', (err) => {
                    console.error('PTY error:', err);
                });
                
                // Store reference for resize events
                socket.ptySocket = ptySocket;
            });

            socket.on('disconnect', () => {
                console.log('WebSocket disconnected');
            });

            socket.on('connect_error', (err) => {
                console.error('WebSocket connection error:', err);
            });

            term.focus();
        }

        // Delete project workspace directory
        async function deleteProject(projectName) {
            if (!confirm(`Are you sure you want to remove the workspace '${projectName}'? (All background terminal sessions for this workspace will be closed).`)) {
                return;
            }

            try {
                const response = await fetch(`/api/projects/${projectName}`, {
                    method: 'DELETE'
                });
                
                if (!response.ok) {
                    const data = await response.json();
                    alert(data.error || "Failed to remove workspace");
                    return;
                }

                projects = await response.json();
                
                if (activeProject === projectName) {
                    activeProject = null;
                    if (socket) {
                        socket.disconnect();
                        socket = null;
                    }
                    document.getElementById('workspace-header').style.display = 'none';
                    document.getElementById('terminal-wrapper').style.display = 'none';
                    document.getElementById('no-workspace-placeholder').style.display = 'flex';
                }

                handleSearch();
            } catch (err) {
                console.error("Error removing workspace:", err);
            }
        }

        // Add Workspace Modal Dialog Controls
        function openAddModal() {
            document.getElementById('proj-name').value = '';
            document.getElementById('proj-path').value = '';
            document.getElementById('modal-error-msg').style.display = 'none';
            document.getElementById('add-modal').classList.add('show');
            document.getElementById('proj-name').focus();
        }

        function closeAddModal() {
            document.getElementById('add-modal').classList.remove('show');
        }

        async function handleAddProject(event) {
            event.preventDefault();
            const errorMsg = document.getElementById('modal-error-msg');
            errorMsg.style.display = 'none';

            const name = document.getElementById('proj-name').value;
            const path = document.getElementById('proj-path').value;

            try {
                const response = await fetch('/api/projects', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ name: name, path: path })
                });

                const data = await response.json();

                if (!response.ok) {
                    errorMsg.innerText = data.error || "Failed to add workspace folder";
                    errorMsg.style.display = 'block';
                    return;
                }

                projects = data;
                handleSearch();
                closeAddModal();
                selectProject(name);
            } catch (err) {
                errorMsg.innerText = "Connection error. Please ensure background server is running.";
                errorMsg.style.display = 'block';
                console.error("Error adding workspace:", err);
            }
        }

            const theme = contextProject.theme || {

                bgColor: '#161c26',
                textColor: '#f1f5f9',
                pathColor: '#94a3b8',
                accentColor: '#2563eb'
            };
            
            document.getElementById('theme-bg').value = theme.background;
            document.getElementById('theme-fg').value = theme.foreground;
            document.getElementById('theme-cursor').value = theme.cursor;
            
            document.getElementById('card-bg').value = cardTheme.bgColor
