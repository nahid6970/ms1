# CODE MERGE FORMAT GUIDE

When I provide code files below, return ALL modifications using ONLY this exact format.
Do NOT write prose explanations outside of comments inside the code blocks.

---

## RESPONSE FORMAT RULES

1. Each changed file starts with `@@FILE:` followed by the relative path.
2. Use `@@MODE: replace_block` to replace a specific block of code.
3. Use `@@MODE: replace_file` to replace the entire file content.
4. Use `@@MODE: insert_after` to insert code after a specific line.
5. Use `@@MODE: delete_block` to delete a block of code.
6. Every change ends with `@@END`.
7. If NO changes needed for a file, omit it entirely.
8. Use project-relative paths in `@@FILE:` entries whenever a project root is provided.
9. Keep the format exact. Some AI tools are more literal, so avoid extra prose between change blocks.

---

## FORMAT TEMPLATES

### Replace a block (most common):
```
@@FILE: path/to/file.py
@@MODE: replace_block
@@FROM:
exact original code here
(must match exactly, including indentation)
@@TO:
new replacement code here
@@END
```

### Replace entire file:
```
@@FILE: path/to/file.py
@@MODE: replace_file
@@TO:
full new file content here
@@END
```

### Insert after a specific line:
```
@@FILE: path/to/file.py
@@MODE: insert_after
@@AFTER:
the exact line to insert after
@@INSERT:
new lines to insert here
@@END
```

### Delete a block:
```
@@FILE: path/to/file.py
@@MODE: delete_block
@@FROM:
exact block to delete
@@END
```

---

## IMPORTANT RULES FOR AI

- The `@@FROM:` block MUST be an exact copy of the original code (same spacing, indentation, line breaks).
- Do NOT add line numbers.
- Do NOT use markdown code fences (``` ``` ```) inside the change blocks.
- You MAY include multiple `@@FILE:` blocks in one response.
- If a file is new (doesn't exist yet), use `@@MODE: replace_file`.

---

## EXAMPLE

User has this in `utils.py`:
```python
def greet(name):
    print("hello " + name)
```

Correct AI response:
```
@@FILE: utils.py
@@MODE: replace_block
@@FROM:
def greet(name):
    print("hello " + name)
@@TO:
def greet(name: str) -> None:
    """Greet the user by name."""
    print(f"Hello, {name}!")
@@END
```

## RESPONSE WRAPPER

Always wrap your **entire response** in a single markdown code block like this:

\`\`\`
@@FILE: ...
...
@@END
\`\`\`

This ensures copy-paste preserves indentation correctly regardless of the web UI used.

### `app.py`
```py
from __future__ import annotations

import json
import threading
import time
from datetime import datetime, timezone
from pathlib import Path
from urllib.request import Request, urlopen

from flask import Flask, Response, jsonify, send_from_directory


BASE_DIR = Path(__file__).resolve().parent
DATA_FILE = BASE_DIR / "groups.json"
ESPN_STANDINGS_URL = "https://site.api.espn.com/apis/v2/sports/soccer/fifa.world/standings"
REFRESH_SECONDS = 60

app = Flask(__name__)

_cache_lock = threading.Lock()
_cache_payload: dict | None = None
_cache_timestamp = 0.0

EMPTY_PAYLOAD = {"updated_at": None, "groups": []}


def _load_disk_payload() -> dict | None:
    if not DATA_FILE.exists():
        return None
    try:
        return json.loads(DATA_FILE.read_text(encoding="utf-8"))
    except Exception:
        return None


def _save_payload(payload: dict) -> None:
    DATA_FILE.write_text(json.dumps(payload, indent=2), encoding="utf-8")


def _normalize_payload(raw: dict) -> dict:
    groups: list[dict] = []
    for grp in raw.get("children", []):
        rows: list[dict] = []
        standings = grp.get("standings", {})
        for entry in standings.get("entries", []):
            stats = {stat.get("abbreviation"): stat.get("displayValue", "0") for stat in entry.get("stats", [])}
            team = entry.get("team", {})
            logos = team.get("logos") or [{}]
            rows.append(
                {
                    "name": team.get("displayName", "Unknown"),
                    "logo": logos[0].get("href", ""),
                    "P": stats.get("GP", "0"),
                    "W": stats.get("W", "0"),
                    "D": stats.get("D", "0"),
                    "L": stats.get("L", "0"),
                    "GD": stats.get("GD", "0"),
                    "Pts": stats.get("P", "0"),
                }
            )

        groups.append({"name": grp.get("name", "Group"), "teams": rows})

    return {
        "updated_at": datetime.now(timezone.utc).isoformat(),
        "groups": groups,
    }


def _refresh_from_espn() -> dict:
    request = Request(ESPN_STANDINGS_URL, headers={"User-Agent": "Mozilla/5.0"})
    with urlopen(request, timeout=20) as response:
        raw = json.load(response)

    payload = _normalize_payload(raw)
    _save_payload(payload)

    global _cache_payload, _cache_timestamp
    with _cache_lock:
        _cache_payload = payload
        _cache_timestamp = time.time()

    return payload


def _get_groups(force: bool = False) -> dict:
    global _cache_payload, _cache_timestamp

    with _cache_lock:
        payload = _cache_payload
        age = time.time() - _cache_timestamp

    if not force and payload is not None and age < REFRESH_SECONDS:
        return payload

    try:
        return _refresh_from_espn()
    except Exception:
        fallback = payload or _load_disk_payload() or EMPTY_PAYLOAD
        with _cache_lock:
            _cache_payload = fallback
            _cache_timestamp = time.time()
        if not DATA_FILE.exists():
            _save_payload(fallback)
        return fallback


def _refresh_loop() -> None:
    while True:
        time.sleep(REFRESH_SECONDS)
        try:
            _refresh_from_espn()
        except Exception:
            pass


@app.route("/")
def index() -> Response:
    return send_from_directory(BASE_DIR, "index.html")


@app.route("/groups.json")
def groups_json() -> Response:
    payload = _get_groups()
    response = jsonify(payload)
    response.headers["Cache-Control"] = "no-store"
    return response


@app.route("/api/groups")
def api_groups() -> Response:
    return groups_json()


@app.route("/api/wins")
def api_wins() -> Response:
    payload = _get_groups()
    teams = []
    for grp in payload.get("groups", []):
        for team in grp.get("teams", []):
            teams.append(
                {
                    "name": team.get("name", "Unknown"),
                    "logo": team.get("logo", ""),
                    "group": grp.get("name", "").replace("Group ", ""),
                    "played": int(team.get("P", "0") or 0),
                    "wins": int(team.get("W", "0") or 0),
                    "draws": int(team.get("D", "0") or 0),
                    "losses": int(team.get("L", "0") or 0),
                    "gd": int(team.get("GD", "0") or 0),
                    "pts": int(team.get("Pts", "0") or 0),
                }
            )
    teams.sort(key=lambda t: (-t["wins"], -t["gd"], -t["pts"], t["name"]))
    response = jsonify({"teams": teams, "updated_at": payload.get("updated_at")})
    response.headers["Cache-Control"] = "no-store"
    return response



def main() -> None:
    try:
        _get_groups(force=True)
    except Exception:
        fallback = _load_disk_payload() or EMPTY_PAYLOAD
        _save_payload(fallback)
        global _cache_payload, _cache_timestamp
        with _cache_lock:
            _cache_payload = fallback
            _cache_timestamp = time.time()

    thread = threading.Thread(target=_refresh_loop, daemon=True)
    thread.start()
    app.run(host="0.0.0.0", port=5152, debug=False, use_reloader=False)


if __name__ == "__main__":
    main()

```

/* ── Live match fetcher ── */
const ESPN = 'https://site.api.espn.com/apis/site/v2/sports/soccer/fifa.world/scoreboard?dates=20260611-20260719';

function bdTime(utcStr) {
  const d = new Date(utcStr);
  d.setHours(d.getHours() + 6); // UTC → BD
  let h = d.getUTCHours(), m = d.getUTCMinutes();
  const ampm = h >= 12 ? 'PM' : 'AM';
  h = h % 12 || 12;
  const mm = m ? ':' + String(m).padStart(2,'0') : '';
  const day = d.toUTCString().slice(5,11); // e.g. "13 Jun"
  return `${h}${mm} ${ampm} · ${day}`;
}

function countdown(utcStr) {
  const diff = new Date(utcStr) - Date.now();
  if (diff <= 0) return '';
  const tot = Math.floor(diff / 1000);
  const d = Math.floor(tot / 86400);
  const h = Math.floor((tot % 86400) / 3600);
  const m = Math.floor((tot % 3600) / 60);
  const parts = [];
  if (d) parts.push(`${d}d`);
  if (h) parts.push(`${h}h`);
  parts.push(`${m}m`);
  return `<div class="kick-countdown">in ${parts.join(' ')}</div>`;
}

function statusBadge(st, clock) {
  const s = st.type.state;
  const name = st.type.name;
  if (s === 'in') {
    if (name === 'STATUS_HALFTIME') return `<span class="badge-ht">HT</span>`;
    let min = '';
    if (clock > 0) {
      const m = Math.floor(clock / 60);
      const sec = Math.floor(clock % 60);
      min = `${m}:${String(sec).padStart(2, '0')}`;
    }
    return `<span class="badge-live">&#9679; LIVE ${min}</span>`;
  }
  if (s === 'post') return `<span class="badge-ft">FT</span>`;
  return `<span class="badge-pre">${st.type.shortDetail||'Upcoming'}</span>`;
}

function scoreOrTime(comp) {
  const st = comp.status;
  const [home, away] = comp.competitors;
  if (st.type.state === 'pre') {
    return `<div class="kick-time">${bdTime(comp.date)}</div>${countdown(comp.date)}`;
  }
  if (st.type.state === 'post') {
    const homeScore = parseInt(home.score, 10) || 0;
    const awayScore = parseInt(away.score, 10) || 0;
    let homeClass = 'score-draw';
    let awayClass = 'score-draw';
    if (homeScore > awayScore) {
      homeClass = 'score-winner';
      awayClass = 'score-loser';
    } else if (awayScore > homeScore) {
      homeClass = 'score-loser';
      awayClass = 'score-winner';
    }
    return `<div class="live-score"><span class="${homeClass}">${home.score}</span><span class="score-dash"> &ndash; </span><span class="${awayClass}">${away.score}</span></div>`;
  }
  const ht = st.type.name === 'STATUS_HALFTIME';
  return `<div class="live-score ${ht?'ht':''}">${home.score} &ndash; ${away.score}</div>`;
}

function goalSide(comp, goal) {
  const homeId = comp.competitors?.[0]?.team?.id;
  const awayId = comp.competitors?.[1]?.team?.id;
  // goal.team = team that scored (beneficiary); athletesInvolved[0].team = player's team
  if (goal.team?.id) {
    return goal.team.id === homeId ? 'home' : goal.team.id === awayId ? 'away' : '';
  }
  // fallback: derive from player's team, then flip for OG
  const involvedId = goal.athletesInvolved?.[0]?.team?.id || '';
  let side = involvedId === homeId ? 'home' : involvedId === awayId ? 'away' : '';
  if (goal.ownGoal) {
    if (side === 'home') side = 'away';
    else if (side === 'away') side = 'home';
  }
  return side;
}

function goalBadge(goal) {
  const min = goal.clock?.displayValue || '';
  const scorers = (goal.athletesInvolved || []).map(a => a.displayName).filter(Boolean);
  const scorerText = scorers.length ? scorers.join(', ') : (goal.shortText || goal.text || '');
  const isGoal = goal.scoringPlay === true;
  const isPenalty = goal.penaltyKick === true;
  const isRed = goal.redCard === true;
  const isYellow = goal.yellowCard === true;
  let icon;
  if (isGoal)        icon = `<span class="goal-icon"></span>`;
  else if (isYellow && isRed) icon = `<span class="card-yr"></span>`;
  else if (isRed)    icon = `<span class="card-r"></span>`;
  else if (isYellow) icon = `<span class="card-y"></span>`;
  else               icon = `<span class="goal-icon"></span>`;
  const suffix = isPenalty ? ' (P)' : '';
  const label = goal.ownGoal ? `OG · ${scorerText}` : `${scorerText}${suffix}`;
  return `<div class="goal-item"><span>${label} <span class="goal-min">${min}</span></span>${icon}</div>`;
}

async function fetchMatches() {
  try {
    const r = await fetch(ESPN);
    const data = await r.json();
    const events = data.events || [];

    // group by date (BD) — only live/finished
    const byDate = {};
    events.forEach(ev => {
      const comp = ev.competitions[0];
      if (comp.status.type.state === 'pre') return;
      const bd = new Date(new Date(comp.date).getTime() + 6*3600000);
      const key = bd.toUTCString().slice(0,16); // "Wed, 11 Jun 2026"
      if (!byDate[key]) byDate[key] = [];
      byDate[key].push({ev, comp});
    });

    let html = '';
    for (const [date, matches] of Object.entries(byDate).sort((a,b) => new Date(b[0]) - new Date(a[0]))) {
      const hasLive = matches.some(m => m.comp.status.type.state === 'in');
      const matchCount = matches.length;
      
      html += `
      <div class="day-group">
        <div class="day-header">
          <div class="day-title">
            ${hasLive ? '<span class="status-live-dot">&#128994;</span>' : '&#128197;'} ${date}
          </div>
          <span class="day-count">${matchCount} ${matchCount === 1 ? 'Match' : 'Matches'}</span>
        </div>
        <div class="day-matches">`;

      for (const {ev, comp} of matches.sort((a,b) => new Date(b.comp.date) - new Date(a.comp.date))) {
        const [home, away] = comp.competitors;
        const st = comp.status;
        const isLive = st.type.state === 'in';
        const cardClass = isLive ? ' card-live' : ' card-past';
        const isGoalEvent = g => g.scoringPlay || g.redCard || g.yellowCard;
        const homeGoals = (comp.details || []).filter(g => isGoalEvent(g) && goalSide(comp, g) === 'home');
        const awayGoals = (comp.details || []).filter(g => isGoalEvent(g) && goalSide(comp, g) === 'away');

        // Identify winner and loser classes for ended matches
        const homeScore = parseInt(home.score, 10) || 0;
        const awayScore = parseInt(away.score, 10) || 0;
        let homeWinnerClass = '';
        let awayWinnerClass = '';
        if (st.type.state === 'post') {
          if (homeScore > awayScore) {
            homeWinnerClass = ' winner';
            awayWinnerClass = ' loser';
          } else if (awayScore > homeScore) {
            homeWinnerClass = ' loser';
            awayWinnerClass = ' winner';
          }
        }

        html += `
        <div class="match-card${cardClass}">
          <div class="team team-panel${homeWinnerClass}">
            <div class="team-head">
              <img class="flag" src="${home.team.logo}" onerror="this.src='https://flagcdn.com/w40/un.png'" alt="${home.team.displayName}">
              <span class="team-name">${home.team.displayName}</span>
            </div>
            ${homeGoals.length ? `<div class="goal-list">${homeGoals.map(goalBadge).join('')}</div>` : ''}
          </div>
          <div class="score-box">
            ${scoreOrTime(comp)}
            <div class="match-meta">${statusBadge(st, st.clock)}</div>
            <div class="match-meta">${[comp.venue?.address?.country, comp.venue?.displayName].filter(Boolean).join(' · ')}</div>
          </div>
          <div class="team team-panel right${awayWinnerClass}">
            <div class="team-head">
              <img class="flag" src="${away.team.logo}" onerror="this.src='https://flagcdn.com/w40/un.png'" alt="${away.team.displayName}">
              <span class="team-name">${away.team.displayName}</span>
            </div>
            ${awayGoals.length ? `<div class="goal-list">${awayGoals.map(goalBadge).join('')}</div>` : ''}
          </div>
        </div>`;
      }

      html += `
        </div>

async function fetchSchedule() {

    const events = data.events || [];

    const byDate = {};
    events.forEach(ev => {

async function fetchScorers() {
  try {
    const r = await fetch(ESPN);
    const data = await r.json();
    const events = data.events || [];

    const scorersMap = {};

    events.forEach(ev => {
      const comp = ev.competitions[0];
      const details = comp.details || [];
      const homeTeam = comp.competitors?.[0]?.team;
      const awayTeam = comp.competitors?.[1]?.team;

      details.forEach(goal => {
        if (goal.scoringPlay === true && goal.ownGoal !== true) {
          let scorerName = '';
          if (goal.athletesInvolved && goal.athletesInvolved.length > 0) {
            scorerName = goal.athletesInvolved[0].displayName || goal.athletesInvolved[0].shortName || '';
          }
          if (!scorerName) {
            scorerName = goal.shortText || goal.text || '';
            scorerName = scorerName.replace(/\s*\d+['+]*\s*$/, '').trim();
          }

          if (scorerName) {
            const side = goalSide(comp, goal);
            const team = side === 'home' ? homeTeam : side === 'away' ? awayTeam : null;
            const teamName = team ? team.displayName : 'Unknown';
            const teamLogo = team ? team.logo : '';

            const key = `${scorerName}__${teamName}`;
            if (!scorersMap[key]) {
              scorersMap[key] = {
                name: scorerName,
                teamName: teamName,
                teamLogo: teamLogo,
                goals: 0
              };
            }
            scorersMap[key].goals += 1;
          }
        }
      });
    });

    const sortedScorers = Object.values(scorersMap).sort((a, b) => {
      if (b.goals !== a.goals) {
        return b.goals - a.goals;
      }
      return a.name.localeCompare(b.name);
    });

    let html = '';
    if (sortedScorers.length === 0) {
      html = '<p style="color:var(--text-muted);padding:30px;text-align:center">No goals scored yet. Check back once matches begin!</p>';
    } else {
      html = '<div class="scorers-list">';
      let rank = 0;
      let prevGoals = -1;
      let displayRank = 0;

      sortedScorers.forEach((scorer, idx) => {
        rank++;
        if (scorer.goals !== prevGoals) {
          displayRank = rank;
          prevGoals = scorer.goals;
        }

        html += `
        <div class="scorer-row">
          <div class="scorer-rank">#${displayRank}</div>
          <div class="scorer-info">
            <div>
              <div class="scorer-name">${scorer.name}</div>
              <div class="scorer-team">
                <img class="flag-sm" src="${scorer.teamLogo}" onerror="this.src='https://flagcdn.com/w40/un.png'" alt="${scorer.teamName}">
                <span>${scorer.teamName}</span>
              </div>
            </div>
          </div>
          <div class="scorer-goals">${scorer.goals} ${scorer.goals === 1 ? 'Goal' : 'Goals'}</div>
        </div>`;
      });
      html += '</div>';
    }

    document.getElementById('scorersContainer').innerHTML = html;
    const t = new Date().toLocaleTimeString('en-BD', {timeZone:'Asia/Dhaka'});
    document.getElementById('scorers-status').textContent = `Updated: ${t} BD`;
  } catch(e) {
    document.getElementById('scorersContainer').innerHTML = '<p style="color:var(--red);padding:20px;text-align:center">Failed to load top scorers.</p>';
  }
}
